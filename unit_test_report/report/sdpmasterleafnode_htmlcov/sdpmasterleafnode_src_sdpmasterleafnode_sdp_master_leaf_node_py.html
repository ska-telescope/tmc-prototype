40 def commandCallback(self, event): 

41 

42 """ 

43 Checks whether the command has been successfully invoked on SDP Master. 

44 

45 :param event: response from SDP Master for the invoked command. 

46 

47 :return: None. 

48 """ 

49 exception_count = 0 

50 exception_message = [] 

51 try: 

52 if event.err: 

53 log = const.ERR_INVOKING_CMD + event.cmd_name 

54 log_msg = const.ERR_INVOKING_CMD + str(event.cmd_name) + "\n" + str(event.errors) 

55 self.logger.error(log_msg) 

56 self._read_activity_message = const.ERR_INVOKING_CMD + str(event.cmd_name) + "\n" + str( 

57 event.errors) 

58 self.logger.error(log) 

59 else: 

60 log = const.STR_COMMAND + event.cmd_name + const.STR_INVOKE_SUCCESS 

61 self._read_activity_message = log 

62 self.logger.info(log) 

63 except Exception as except_occurred: 

64 self._read_activity_message = const.ERR_EXCEPT_CMD_CB + str(except_occurred) 

65 log_msg = const.ERR_EXCEPT_CMD_CB + str(except_occurred) 

66 self.logger.error(log_msg) 

67 exception_message.append(self._read_activity_message) 

68 exception_count += 1 

69 

70 # Throw Exception 

71 if exception_count > 0: 

72 err_msg = '' 

73 for item in exception_message: 

74 err_msg += item + "\n" 

75 tango.Except.throw_exception(const.STR_CMD_FAILED, err_msg, 

76 const.STR_SDP_CMD_CALLBK, tango.ErrSeverity.ERR) 

77 

78 #Throw devfailed exception 

79 def _handle_devfailed_exception(self, df, actvity_msg): 

80 self._read_activity_message = actvity_msg 

81 self._read_activity_message = const.ERR_MSG + str(df) 

82 self.logger.error(actvity_msg)
