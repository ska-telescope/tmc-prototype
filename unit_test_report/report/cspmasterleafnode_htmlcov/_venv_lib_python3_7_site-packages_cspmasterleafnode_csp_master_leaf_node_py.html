143 def commandCallback(self, event): 

144 """ 

145 Checks whether the command has been successfully invoked on CSPMaster. 

146 

147 :param event: response from CspMaster for the invoked command. 

148 

149 :return: None 

150 """ 

151 exception_count = 0 

152 exception_message = [] 

153 try: 

154 if event.err: 

155 log = const.ERR_INVOKING_CMD + event.cmd_name 

156 log_msg = const.ERR_INVOKING_CMD + str(event.cmd_name) + "\n" + str(event.errors) 

157 self.logger.error(log_msg) 

158 self._read_activity_message = const.ERR_INVOKING_CMD + str(event.cmd_name) + "\n" + str( 

159 event.errors) 

160 else: 

161 log = const.STR_COMMAND + event.cmd_name + const.STR_INVOKE_SUCCESS 

162 self._read_activity_message = log 

163 except Exception as except_occurred: 

164 self._handle_generic_exception(const.ERR_EXCEPT_CMD_CB + ": " + str(except_occurred)) 

165 exception_message.append(self._read_activity_message) 

166 exception_count += 1 

167 

168 # Throw Exception 

169 if exception_count > 0: 

170 err_msg = '' 

171 for item in exception_message: 

172 err_msg += item + "\n" 

173 tango.Except.throw_exception(const.STR_CMD_FAILED, err_msg, 

174 const.STR_CSP_CMD_CALLBK, tango.ErrSeverity.ERR)
