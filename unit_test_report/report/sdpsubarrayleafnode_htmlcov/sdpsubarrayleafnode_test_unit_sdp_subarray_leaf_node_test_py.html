344@contextlib.contextmanager 

345def fake_tango_system(device_under_test, initial_dut_properties={}, proxies_to_mock={}, 

346 device_proxy_import_path='tango.DeviceProxy'): 

347 

348 with mock.patch(device_proxy_import_path) as patched_constructor: 

349 patched_constructor.side_effect = lambda device_fqdn: proxies_to_mock.get(device_fqdn, Mock()) 

350 patched_module = importlib.reload(sys.modules[device_under_test.__module__]) 

351 

352 device_under_test = getattr(patched_module, device_under_test.__name__) 

353 

354 device_test_context = DeviceTestContext(device_under_test, properties=initial_dut_properties) 

355 device_test_context.start() 

356 yield device_test_context 

357 device_test_context.stop()
